## 1.1 `remove_item` → ItemRepo

**文件**：`2workbench/feature/ai/tools.py`

**当前**：
```python
@tool
def remove_item(item_name: str, player_id: int = 1) -> str:
    """从玩家背包移除物品"""
    return f"已从玩家背包移除 {item_name}"
```

**修复**：
```python
@tool
def remove_item(item_name: str, player_id: int = 1) -> str:
    """从玩家背包移除物品"""
    ctx = get_tool_context()
    if not ctx:
        return f"移除失败: 无工具上下文"
    try:
        from core.models.repository import PlayerRepo
        repo = PlayerRepo(db_path=ctx.db_path)
        inventory = repo.get_inventory(player_id)
        for pi in inventory:
            if pi.item and pi.item.item_name == item_name:
                repo.remove_item(player_id, pi.item.id)
                return f"已从背包移除 {item_name}"
        return f"未找到物品: {item_name}"
    except Exception as e:
        return f"移除失败: {e}"
```

## 1.2 `update_npc_relationship` → NPCRepo

**当前**：
```python
@tool
def update_npc_relationship(npc_name: str, change: float, player_id: int = 1) -> str:
    """更新与NPC的关系值"""
    return f"与 {npc_name} 的关系值变化了 {change:+.1f}"
```

**修复**：
```python
@tool
def update_npc_relationship(npc_name: str, change: float, player_id: int = 1) -> str:
    """更新与NPC的关系值（-1.0 到 1.0）"""
    ctx = get_tool_context()
    if not ctx:
        return f"更新失败: 无工具上下文"
    try:
        from core.models.repository import NPCRepo
        repo = NPCRepo(db_path=ctx.db_path)
        npcs = repo.list_all(player_id)  # 或按 world_id 查询
        for npc in npcs:
            if npc.name == npc_name:
                new_rel = max(-1.0, min(1.0, npc.relationship + change))
                repo.update(npc.id, relationship=new_rel)
                return f"与 {npc_name} 的关系值: {npc.relationship:.1f} → {new_rel:.1f}"
        return f"未找到 NPC: {npc_name}"
    except Exception as e:
        return f"更新失败: {e}"
```

## 1.3 `update_quest_status` → QuestRepo

**当前**：
```python
@tool
def update_quest_status(quest_id: int, status: str) -> str:
    """更新任务状态"""
    return f"任务 {quest_id} 状态已更新为 {status}"
```

**修复**：
```python
@tool
def update_quest_status(quest_id: int, status: str) -> str:
    """更新任务状态（active/completed/failed/abandoned）"""
    ctx = get_tool_context()
    if not ctx:
        return f"更新失败: 无工具上下文"
    valid = ("active", "completed", "failed", "abandoned")
    if status not in valid:
        return f"无效状态: {status}，可选: {valid}"
    try:
        from core.models.repository import QuestRepo
        repo = QuestRepo(db_path=ctx.db_path)
        quest = repo.get_by_id(quest_id)
        if not quest:
            return f"未找到任务: {quest_id}"
        repo.update(quest_id, status=status)
        return f"任务「{quest.title}」状态: {quest.status} → {status}"
    except Exception as e:
        return f"更新失败: {e}"
```

## 1.4 `store_memory` → MemoryRepo

**当前**：
```python
@tool
def store_memory(content: str, category: str = "session", importance: float = 0.5) -> str:
    """存储记忆"""
    return f"已存储记忆: {content[:30]}..."
```

**修复**：
```python
@tool
def store_memory(content: str, category: str = "session", importance: float = 0.5, player_id: int = 1) -> str:
    """存储记忆到数据库（category: session/character/world/event）"""
    ctx = get_tool_context()
    if not ctx:
        return f"存储失败: 无工具上下文"
    try:
        from core.models.repository import MemoryRepo
        repo = MemoryRepo(db_path=ctx.db_path)
        memory = repo.create(
            world_id=ctx.world_id,
            player_id=player_id,
            content=content,
            category=category,
            importance=max(0.0, min(1.0, importance)),
        )
        return f"已存储记忆 (id={memory.id}, category={category})"
    except Exception as e:
        return f"存储失败: {e}"
```
