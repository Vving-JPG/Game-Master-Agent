下面我会提供一份优化后的实现，同时解决了以下问题：
- 取消测试后不会再弹出结果窗口  
- 避免重复点击导致多个测试线程同时运行  
- 线程安全退出，不再使用 `terminate()`  
- 改进的 `base_url` 推断（可扩展且容易维护）  
- 更清晰的错误提示

### ✅ 优化后的 `TestApiWorker`（替换原有类）

```python
import threading
import asyncio
from PyQt6.QtCore import QThread, pyqtSignal
from foundation.llm import OpenAICompatibleClient, LLMMessage
from foundation.logger import get_logger

logger = get_logger(__name__)

# 将 base_url 推断抽取为独立函数，方便维护
def _infer_base_url(model_name: str) -> str:
    """根据模型名返回默认 API 地址，找不到返回空字符串"""
    mapping = {
        "deepseek": "https://api.deepseek.com",
        "glm": "https://open.bigmodel.cn/api/paas/v4",
        "qwen": "https://dashscope.aliyuncs.com/compatible-mode/v1",
        # 可继续扩展
    }
    lower = model_name.lower()
    for key, url in mapping.items():
        if key in lower:
            return url
    return ""


class TestApiWorker(QThread):
    """在后台线程中测试 API 连接，支持安全取消"""
    test_finished = pyqtSignal(bool, str, str)  # success, message, model_name

    def __init__(self, model: str, api_key: str, base_url: str = ""):
        super().__init__()
        self.model = model
        self.api_key = api_key
        self.base_url = base_url or _infer_base_url(model)
        self._stop_event = threading.Event()      # 线程安全取消标志

    def cancel(self):
        """通知线程停止（不会立即中断异步任务，但会尽快结束）"""
        self._stop_event.set()

    def run(self):
        try:
            if self._stop_event.is_set():
                self.test_finished.emit(False, "测试已取消", self.model)
                return

            # 创建新的事件循环
            loop = asyncio.new_event_loop()
            asyncio.set_event_loop(loop)

            client = OpenAICompatibleClient(
                provider_name="test",
                api_key=self.api_key,
                base_url=self.base_url if self.base_url else None,
                model=self.model,
                max_retries=1,
                timeout=15,          # 单次请求超时
            )

            async def do_test():
                # 每隔0.2秒检查一次取消标志，支持提前退出
                for _ in range(100):      # 最长约20秒
                    if self._stop_event.is_set():
                        return None
                    await asyncio.sleep(0.2)
                # 实际请求（无法中断底层 socket，但线程可安全退出）
                response = await client.chat_async(
                    messages=[LLMMessage(role="user", content="Hello, reply with 'OK'")],
                    temperature=0.1,
                    max_tokens=50,
                )
                return response

            response = loop.run_until_complete(do_test())
            loop.close()

            if self._stop_event.is_set():
                self.test_finished.emit(False, "测试已取消", self.model)
                return

            if not response:
                self.test_finished.emit(False, "API 无响应或测试被中断", self.model)
                return

            # 成功判断：有 token 消耗
            if response.completion_tokens > 0 or response.prompt_tokens > 0:
                preview = response.content[:100] if response.content else "(空内容)"
                msg = (f"响应: {preview}\n"
                       f"finish_reason: {response.finish_reason}, "
                       f"tokens: {response.prompt_tokens}/{response.completion_tokens}")
                self.test_finished.emit(True, msg, self.model)
            else:
                self.test_finished.emit(
                    False,
                    f"API 返回异常 (finish_reason: {response.finish_reason}, "
                    f"tokens: {response.prompt_tokens}/{response.completion_tokens})",
                    self.model
                )

        except Exception as e:
            if self._stop_event.is_set():
                self.test_finished.emit(False, "测试已取消", self.model)
            else:
                import traceback
                detail = f"{str(e)[:150]}\n{traceback.format_exc()[-200:]}"
                self.test_finished.emit(False, detail, self.model)
        finally:
            self._stop_event.clear()   # 清理状态
```

### ✅ 优化后的 `ModelListWidget` 测试相关方法

主要改动：防止重复测试、取消后不弹窗、清晰的状态管理。

```python
def _on_test(self, model_id: str) -> None:
    """测试 API 连接（使用后台线程避免阻塞 UI）"""
    # 如果已有测试正在进行，先取消
    if hasattr(self, '_test_worker') and self._test_worker.isRunning():
        self._test_worker.cancel()
        self._test_worker.wait(2000)

    data = next((m for m in self._custom_models if m.get("id") == model_id), None)
    if not data:
        QMessageBox.warning(self, "测试失败", "找不到模型配置")
        return

    model_name = data.get("model", "")
    api_key = data.get("api_key", "")
    base_url = data.get("base_url", "")

    if not api_key:
        QMessageBox.warning(self, "测试失败", "API 密钥为空")
        return

    # 创建非阻塞的提示框
    self._test_dialog = QMessageBox(self)
    self._test_dialog.setWindowTitle("测试中")
    self._test_dialog.setText("正在测试 API 连接，请稍候...\n(点击取消中止测试)")
    self._test_dialog.setStandardButtons(QMessageBox.StandardButton.Cancel)
    self._test_dialog.setEscapeButton(QMessageBox.StandardButton.Cancel)

    # 连接取消信号
    self._test_dialog.rejected.connect(self._on_test_cancelled)

    # 创建后台测试线程
    self._test_worker = TestApiWorker(model_name, api_key, base_url)
    self._test_worker.test_finished.connect(self._on_test_finished)

    # 开始测试并显示对话框
    self._test_worker.start()
    self._test_dialog.open()   # 非模态，不阻塞主界面


def _on_test_cancelled(self) -> None:
    """用户点击取消按钮或关闭对话框"""
    if hasattr(self, '_test_worker') and self._test_worker:
        self._test_worker.cancel()
        # 不必等待线程完全结束，finished 信号会处理对话框关闭
    # 将对话框置空，防止重复关闭
    self._test_dialog = None


def _on_test_finished(self, success: bool, message: str, model_name: str) -> None:
    """测试完成后的统一处理"""
    # 如果对话框已被用户关闭，不再弹出新窗口（取消时 _test_dialog 已被设为 None）
    if not hasattr(self, '_test_dialog') or self._test_dialog is None:
        logger.debug("测试对话框已关闭，忽略结果")
        return

    # 断开取消信号，避免重复触发
    try:
        self._test_dialog.rejected.disconnect(self._on_test_cancelled)
    except TypeError:
        pass

    # 关闭等待框
    self._test_dialog.close()
    self._test_dialog = None

    # 显示结果
    if success:
        QMessageBox.information(self, "测试成功",
                                f"模型: {model_name}\n{message}\n\nAPI 连接正常！")
        logger.info(f"API 测试成功: {model_name}")
    else:
        QMessageBox.critical(self, "测试失败",
                             f"模型: {model_name}\n错误: {message}\n\n请检查密钥和地址")
        logger.error(f"API 测试失败: {model_name}, 错误: {message}")
```

### 💡 配套建议

1. **持久化模型 ID**（解决无法编辑的 bug）  
   在 `_load_custom_models` 和 `_sync_to_config` 中将 `id` 存入 JSON，例如：
   ```json
   "providers": {
     "deepseek": {
       "id": "uuid-xxxx",
       "api_key": "...",
       "base_url": "...",
       "model": "deepseek-chat"
     }
   }
   ```
   加载时如果已有 `id` 则复用，否则生成新 UUID 并写回。

2. **让 `base_url` 推断更透明**  
   在 `AddModelDialog` 中，当用户输入模型名后，可以预填默认地址（根据 `_infer_base_url`），并允许直接修改，避免用户不知道实际请求的地址。

3. **去掉 `_save` 方法的冗余**  
   现在 `_save` 只是调用 `_sync_to_config`，可以直接删除，调用处直接使用 `_sync_to_config`。
