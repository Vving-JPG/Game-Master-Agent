## 2.1 Prompt 编辑器持久化到文件

**文件**：`2workbench/presentation/editor/prompt_editor.py:218`

**当前问题**：
`_save_prompt` 只将内容保存到内存 `self._prompts` 字典，不写入文件。关闭项目后修改丢失。

**修复方案**：

```python
class PromptEditorWidget(BaseWidget):
    def _save_prompt(self) -> None:
        """保存 Prompt 到内存 + 持久化到文件"""
        key = self._current_prompt_key
        if not key:
            return

        content = self._editor.toPlainText()
        if not content.strip():
            return

        # 1. 保存到内存
        if key not in self._prompts:
            self._prompts[key] = PromptVersion(key=key, versions=[])
        self._prompts[key].versions.append(PromptVersionEntry(
            content=content,
            timestamp=datetime.now().isoformat(),
        ))

        # 2. 持久化到文件（通过 project_manager）
        try:
            from presentation.project.manager import project_manager
            project_manager.save_prompt(key, content)
            self._status_label.setText(
                f"已保存: {key} (v{len(self._prompts[key].versions)})"
            )
            self._logger.info(f"Prompt 已持久化: {key}")
        except RuntimeError as e:
            self._status_label.setText(f"保存失败: {e}")
            self._logger.error(f"Prompt 持久化失败: {e}")
```

**同时在主窗口中确保 Prompt 加载时从文件读取**：

```python
# main_window.py — 在 _do_open_project 中
def _do_open_project(self, project_path: str):
```

## 2.2 图编辑器节点删除 Bug 修复

**文件**：`2workbench/presentation/editor/graph_editor.py:318-330`

**当前问题**：
`GraphScene.remove_node` 遍历关联边但只做 `pass`，`GraphEdgeItem` 未从场景中移除，导致图形残留。`GraphNodeItem._delete` 只调用 `scene.removeItem(self)` 但未调用 `scene.remove_node`。

**修复方案**：

```python
class GraphScene(QGraphicsScene):
    def remove_node(self, node_id: str) -> bool:
        """删除节点及其所有关联边"""
        if node_id not in self._nodes:
            return False

        node = self._nodes.pop(node_id)

        # 删除所有关联的边（从场景中移除 QGraphicsItem）
        edges_to_remove = [e for e in self._edges
                           if e.source.node_id == node_id or e.target.node_id == node_id]
        for edge in edges_to_remove:
            self._edges.remove(edge)
            self.removeItem(edge)  # ← 关键：从场景中移除
            logger.debug(f"删除边: {edge.source.node_id} -> {edge.target.node_id}")

        # 删除节点
        self.removeItem(node)  # ← 关键：从场景中移除
        logger.debug(f"删除节点: {node_id}")
        return True
```

**同时修复 `GraphNodeItem._delete`**：

```python
class GraphNodeItem(QGraphicsGraphicsObject):
    def _delete(self) -> None:
        """删除此节点"""
        if self.scene():
            self.scene().remove_node(self.node_id)  # ← 使用 scene 的方法
```

## 2.3 图编辑器拖拽连线交互

**文件**：`2workbench/presentation/editor/graph_editor.py`

**当前问题**：
只能通过右键菜单"连接到..."来连线，缺少可视化拖拽连线交互。

**修复方案**：

在 `GraphEditorView` 中添加拖拽连线：

```python
class GraphEditorView(QGraphicsView):
    def __init__(self, scene: GraphScene):
        super().__init__(scene)
        self._dragging_edge = False
        self._drag_source_port = None  # (node_id, "output"|"input")
        self._temp_line: QGraphicsLineItem | None = None
        self.setDragMode(QGraphicsView.DragMode.NoDrag)
        self.setRenderHint(QPainter.RenderHint.Antialiasing)

    def mousePressEvent(self, event: QMouseEvent):
        """检测是否点击了连接点"""
        if event.button() == Qt.MouseButton.LeftButton:
            item = self.itemAt(event.pos())
            if isinstance(item, QGraphicsEllipseItem) and item.parentItem():
                node = item.parentItem()
                if isinstance(node, GraphNodeItem):
                    # 判断是输入还是输出端口
                    port_pos = item.pos()
                    node_center = node.boundingRect().center()
                    if port_pos.x() > node_center.x():
                        self._drag_source_port = (node.node_id, "output")
                    else:
                        self._drag_source_port = (node.node_id, "input")
                    self._dragging_edge = True
                    return  # 不传递给基类
        super().mousePressEvent(event)

    def mouseMoveEvent(self, event: QMouseEvent):
        """拖拽时绘制临时连线"""
        if self._dragging_edge and self._drag_source_port:
            scene_pos = self.mapToScene(event.pos())

            if self._temp_line is None:
                self._temp_line = QGraphicsLineItem()
                self._temp_line.setPen(QPen(QColor("#4A90D9"), 2, Qt.PenStyle.DashLine))
                self.scene().addItem(self._temp_line)

            # 获取源端口位置
            source_node = self.scene()._nodes.get(self._drag_source_port[0])
            if source_node:
                if self._drag_source_port[1] == "output":
                    start = source_node.get_output_pos()
                else:
                    start = source_node.get_input_pos()
                self._temp_line.setLine(start.x(), start.y(), scene_pos.x(), scene_pos.y())
            return
        super().mouseMoveEvent(event)

    def mouseReleaseEvent(self, event: QMouseEvent):
        """释放时连接到目标节点"""
        if self._dragging_edge and self._drag_source_port:
            # 清除临时线
            if self._temp_line:
                self.scene().removeItem(self._temp_line)
                self._temp_line = None

            # 检测目标节点
            target_item = self.itemAt(event.pos())
            if isinstance(target_item, QGraphicsEllipseItem) and target_item.parentItem():
                target_node = target_item.parentItem()
                if isinstance(target_node, GraphNodeItem):
                    source_id = self._drag_source_port[0]
                    target_id = target_node.node_id
                    if source_id != target_id:
                        self.scene().add_edge(source_id, target_id)
                        logger.debug(f"拖拽连线: {source_id} -> {target_id}")

            self._dragging_edge = False
            self._drag_source_port = None
            return
        super().mouseReleaseEvent(event)
```

## 2.4 节点 ID 自动生成

**文件**：`2workbench/presentation/editor/graph_editor.py:525-535`

**当前问题**：
`_add_node_dialog` 中节点 ID 固定为 `"new_node"`，多次添加会冲突。

**修复方案**：

```python
class GraphEditorWidget(QWidget):
    _node_counter = 0  # 类变量，自动递增

    def _add_node_dialog(self) -> None:
        """添加节点对话框"""
        dialog = AddNodeDialog(self)
        if dialog.exec() == QDialog.DialogResult.Accepted:
            node_type = dialog.node_type
            label = dialog.label

            # 自动生成唯一 ID
            GraphEditorWidget._node_counter += 1
            node_id = f"{node_type}_{GraphEditorWidget._node_counter}"

            # 检查 ID 是否已存在
            while node_id in self._scene._nodes:
                GraphEditorWidget._node_counter += 1
                node_id = f"{node_type}_{GraphEditorWidget._node_counter}"

            self._scene.add_node(
                node_id=node_id,
                node_type=node_type,
                label=label,
                position={"x": 200 + GraphEditorWidget._node_counter * 30,
                          "y": 200 + GraphEditorWidget._node_counter * 20},
            )
            self._logger.info(f"添加节点: {node_id} ({node_type})")
```
