## Step 7.2: 实现文件编辑和保存

def _open_text_editor(self, file_path: str) -> None:
    """在文本编辑器中打开文件"""
    from pathlib import Path
    from PyQt6.QtWidgets import QTextEdit

    path = Path(file_path)

    # 检查是否已打开
    for i in range(self.center_panel.tab_widget.count()):
        widget = self.center_panel.tab_widget.widget(i)
        if hasattr(widget, '_file_path') and widget._file_path == file_path:
            self.center_panel.tab_widget.setCurrentIndex(i)
            return

    # 创建编辑器
    editor = QTextEdit()
    editor._file_path = file_path
    editor._is_modified = False

    # 读取文件
    try:
        with open(file_path, 'r', encoding='utf-8') as f:
            content = f.read()
        editor.setPlainText(content)
    except Exception as e:
        editor.setPlainText(f"读取文件失败: {e}")

    # 设为可编辑
    editor.setReadOnly(False)

    # 监听修改
    editor.textChanged.connect(self._on_editor_text_changed)

    # 添加标签页
    tab_name = f"📄 {path.name}"
    index = self.center_panel.tab_widget.addTab(editor, tab_name)
    self.center_panel.tab_widget.setCurrentIndex(index)

def _on_editor_text_changed(self) -> None:
    """编辑器内容变化时更新标签页标题"""
    editor = self.center_panel.tab_widget.currentWidget()
    if hasattr(editor, '_file_path') and not editor.isReadOnly():
        if not editor._is_modified:
            editor._is_modified = True
            from pathlib import Path
            name = Path(editor._file_path).name
            index = self.center_panel.tab_widget.indexOf(editor)
            self.center_panel.tab_widget.setTabText(index, f"📄 {name} *")

def _save_current_editor(self) -> None:
    """保存当前编辑器中的文件"""
    editor = self.center_panel.tab_widget.currentWidget()
    if not hasattr(editor, '_file_path') or editor.isReadOnly():
        return

    try:
        with open(editor._file_path, 'w', encoding='utf-8') as f:
            f.write(editor.toPlainText())
        editor._is_modified = False
        from pathlib import Path
        name = Path(editor._file_path).name
        index = self.center_panel.tab_widget.indexOf(editor)
        self.center_panel.tab_widget.setTabText(index, f"📄 {name}")
        self._show_message(f"已保存: {name}")
    except Exception as e:
        QMessageBox.critical(self, "保存失败", str(e))
```

# 修改 CenterPanel._on_tab_close
def _on_tab_close(self, index: int) -> None:
    """关闭标签页（带未保存检查）"""
    if self.tab_widget.count() <= 1:
        return

    widget = self.tab_widget.widget(index)

    # 检查是否有未保存的修改
    if hasattr(widget, '_is_modified') and widget._is_modified:
        from pathlib import Path
        name = Path(widget._file_path).name if hasattr(widget, '_file_path') else "文件"
        reply = QMessageBox.question(
            self, "未保存的更改",
            f"'{name}' 有未保存的更改。\n确定要关闭吗？",
            QMessageBox.StandardButton.Save |
            QMessageBox.StandardButton.Discard |
            QMessageBox.StandardButton.Cancel
        )
        if reply == QMessageBox.StandardButton.Save:
            # 触发保存
            self.window()._save_current_editor()
        elif reply == QMessageBox.StandardButton.Cancel:
            return  # 不关闭

    self.tab_widget.removeTab(index)

    # 清理引用
    if widget == self._graph_editor:
        self._graph_editor = None
    elif widget == self._prompt_editor:
        self._prompt_editor = None
    elif widget == self._tool_manager:
        self._tool_manager = None
```

def _on_run_agent(self) -> None:
    """运行 Agent"""
    from presentation.project.manager import project_manager
    from foundation.config import settings

    if not project_manager.is_open:
        self._show_message("请先打开一个项目")
        return

    if not settings.deepseek_api_key:
        QMessageBox.warning(self, "配置缺失",
            "请先配置 API Key（设置 > API设置）")
        self._on_settings()
        return

    # 编译图
    try:
        from feature.ai.graph_compiler import graph_compiler
        graph_data = project_manager.load_graph()
        compiled = graph_compiler.compile(graph_data)
    except Exception as e:
        QMessageBox.critical(self, "编译失败", f"图编译失败: {e}")
        return

    # 创建 Agent 并运行
    try:
        from feature.ai.gm_agent import GMAgent
        agent = GMAgent(world_id=1)
        agent.set_graph(compiled, source="json")

        # 使用 qasync 异步运行
        import asyncio
        from foundation.event_bus import event_bus

        async def run_agent():
            try:
                result = await agent.run("开始游戏")
                self._show_message("Agent 运行完成")
            except Exception as e:
                logger.error(f"Agent 运行失败: {e}")
                self._show_message(f"Agent 运行失败: {e}")

        # 获取 qasync 事件循环
        import qasync
        loop = qasync.QEventLoop(QApplication.instance())
        asyncio.set_event_loop(loop)
        asyncio.ensure_future(run_agent())

        self._show_message("Agent 运行中...")
    except Exception as e:
        QMessageBox.critical(self, "运行失败", str(e))

## Step 7.5: 实现命令面板（Ctrl+Shift+P）

**问题**: KEY-005 — 缺少命令面板

**文件**: 新建 `2workbench/presentation/widgets/command_palette.py`

**修改方案**:

```python
"""命令面板 — Ctrl+Shift+P 快速执行命令"""
from PyQt6.QtWidgets import (
    QWidget, QVBoxLayout, QListWidget, QListWidgetItem,
    QLineEdit, QFrame,
)
from PyQt6.QtCore import pyqtSignal, Qt


class CommandPalette(QWidget):
    """命令面板"""

    command_executed = pyqtSignal(str)  # 命令 ID

    COMMANDS = [
        ("file.new", "新建项目", "Ctrl+N"),
        ("file.open", "打开项目", "Ctrl+O"),
        ("file.save", "保存", "Ctrl+S"),
        ("view.sidebar", "切换侧边栏", "Ctrl+B"),
        ("view.fullscreen", "全屏", "F11"),
        ("agent.run", "运行 Agent", "F5"),
        ("agent.stop", "停止 Agent", "Shift+F5"),
        ("settings.open", "打开设置", "Ctrl+,"),
        ("theme.dark", "切换 Dark 主题", ""),
        ("theme.light", "切换 Light 主题", ""),
    ]

    def __init__(self, parent=None):
        super().__init__(parent)
        self.setWindowFlags(Qt.WindowType.Popup | Qt.WindowType.FramelessWindowHint)
        self.setAttribute(Qt.WidgetAttribute.WA_TranslucentBackground)
        self._setup_ui()

    def _setup_ui(self):
        layout = QVBoxLayout(self)
        layout.setContentsMargins(10, 10, 10, 10)

        # 搜索框
        self._search = QLineEdit()
        self._search.setPlaceholderText("输入命令...")
        self._search.textChanged.connect(self._filter_commands)
        layout.addWidget(self._search)

        # 命令列表
        self._list = QListWidget()
        self._list.itemDoubleClicked.connect(self._on_item_double_clicked)
        layout.addWidget(self._list)

        self._load_commands()

    def _load_commands(self):
        self._list.clear()
        for cmd_id, label, shortcut in self.COMMANDS:
            text = f"{label}" + (f"  ({shortcut})" if shortcut else "")
            item = QListWidgetItem(text)
            item.setData(Qt.ItemDataRole.UserRole, cmd_id)
            self._list.addItem(item)

    def _filter_commands(self, keyword: str):
        keyword = keyword.lower()
        for i in range(self._list.count()):
            item = self._list.item(i)
            match = keyword == "" or keyword in item.text().lower()
            self._list.setItemHidden(item, not match)

    def _on_item_double_clicked(self, item):
        cmd_id = item.data(Qt.ItemDataRole.UserRole)
        self.command_executed.emit(cmd_id)
        self.hide()

    def show_at(self, pos):
        """在指定位置显示"""
        self.move(pos)
        self._search.clear()
        self._search.setFocus()
        self._load_commands()
        self.show()
```
