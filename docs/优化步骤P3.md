```python
class ItemEditor(QWidget):
    """物品编辑器"""

    def __init__(self, parent=None):
        super().__init__(parent)
        self._items = []
        self._setup_ui()

    def _setup_ui(self):
        layout = QHBoxLayout(self)

        # 左侧列表
        self._table = QTableWidget()
        self._table.setColumnCount(4)
        self._table.setHorizontalHeaderLabels(["ID", "名称", "类型", "描述"])
        self._table.horizontalHeader().setStretchLastSection(True)
        self._table.setSelectionBehavior(QTableWidget.SelectionBehavior.SelectRows)
        self._table.currentCellChanged.connect(self._on_item_selected)
        layout.addWidget(self._table, stretch=2)

        # 右侧编辑面板
        form = QFormLayout()
        self._name_edit = QLineEdit()
        self._type_combo = QComboBox()
        self._type_combo.addItems(["武器", "防具", "消耗品", "任务物品", "材料", "其他"])
        self._desc_edit = QTextEdit()
        self._desc_edit.setMaximumHeight(100)
        self._effect_edit = QTextEdit()
        self._effect_edit.setMaximumHeight(60)
        self._effect_edit.setPlaceholderText("物品效果描述（可选）")

        form.addRow("名称:", self._name_edit)
        form.addRow("类型:", self._type_combo)
        form.addRow("描述:", self._desc_edit)
        form.addRow("效果:", self._effect_edit)

        # 按钮
        btn_layout = QHBoxLayout()
        add_btn = StyledButton("➕ 添加", style_type="primary")
        add_btn.clicked.connect(self._add_item)
        save_btn = StyledButton("💾 保存修改", style_type="secondary")
        save_btn.clicked.connect(self._save_item)
        del_btn = StyledButton("🗑️ 删除", style_type="danger")
        del_btn.clicked.connect(self._delete_item)
        btn_layout.addWidget(add_btn)
        btn_layout.addWidget(save_btn)
        btn_layout.addWidget(del_btn)
        form.addRow(btn_layout)

        layout.addLayout(form, stretch=1)

    def _add_item(self) -> None:
        name = self._name_edit.text().strip()
        if not name:
            QMessageBox.warning(self, "提示", "物品名称不能为空")
            return
        item = {
            "id": len(self._items) + 1,
            "name": name,
            "type": self._type_combo.currentText(),
            "description": self._desc_edit.toPlainText(),
            "effect": self._effect_edit.toPlainText(),
        }
        self._items.append(item)
        self._refresh_table()

    def _save_item(self) -> None:
        row = self._table.currentRow()
        if row < 0:
            return
        self._items[row]["name"] = self._name_edit.text()
        self._items[row]["type"] = self._type_combo.currentText()
        self._items[row]["description"] = self._desc_edit.toPlainText()
        self._items[row]["effect"] = self._effect_edit.toPlainText()
        self._refresh_table()

    def _delete_item(self) -> None:
        row = self._table.currentRow()
        if row < 0:
            return
        reply = QMessageBox.question(self, "确认删除", f"确定删除物品 '{self._items[row]['name']}'？",
            QMessageBox.StandardButton.Yes | QMessageBox.StandardButton.No)
        if reply == QMessageBox.StandardButton.Yes:
            self._items.pop(row)
            self._refresh_table()

    def _refresh_table(self) -> None:
        self._table.setRowCount(len(self._items))
        for i, item in enumerate(self._items):
            self._table.setItem(i, 0, QTableWidgetItem(str(item["id"])))
            self._table.setItem(i, 1, QTableWidgetItem(item["name"]))
            self._table.setItem(i, 2, QTableWidgetItem(item["type"]))
            self._table.setItem(i, 3, QTableWidgetItem(item.get("description", "")[:50]))

    def _on_item_selected(self, row: int, col: int, prev_row: int, prev_col: int) -> None:
        if 0 <= row < len(self._items):
            item = self._items[row]
            self._name_edit.setText(item["name"])
            self._type_combo.setCurrentText(item["type"])
            self._desc_edit.setPlainText(item.get("description", ""))
            self._effect_edit.setPlainText(item.get("effect", ""))

    def get_data(self) -> list:
        return self._items

    def load_data(self, items: list) -> None:
        self._items = items
        self._refresh_table()
```

```python
def _search_in_log(self, keyword: str) -> None:
    """在日志中搜索关键词"""
    if not keyword:
        return
    cursor = self._log_browser.textCursor()
    document = self._log_browser.document()
    found = document.find(keyword, cursor)
    if not found.isNull():
        self._log_browser.setTextCursor(found)
        self._log_browser.ensureCursorVisible()
    else:
        cursor.movePosition(cursor.MoveOperation.Start)
        found = document.find(keyword, cursor)
        if not found.isNull():
            self._log_browser.setTextCursor(found)
```

## Step 3.6: 编排器添加链有效性验证

**问题**: F-016

**修改方案**:

```python
def _validate_chain(self) -> list[str]:
    """验证链的完整性，返回错误列表"""
    errors = []
    agent_ids = {a.id for a in self._agents}

    # 检查孤立 Agent（没有连接的）
    connected_ids = set()
    for link in self._links:
        connected_ids.add(link.agent_id)
        connected_ids.add(link.next_agent_id)
    for agent in self._agents:
        if agent.id not in connected_ids and len(self._agents) > 1:
            errors.append(f"Agent '{agent.name}' 没有任何连接")

    # 检查悬空连接
    for link in self._links:
        if link.agent_id not in agent_ids:
            errors.append(f"连接引用了不存在的 Agent: {link.agent_id}")
        if link.next_agent_id not in agent_ids:
            errors.append(f"连接引用了不存在的 Agent: {link.next_agent_id}")

    # 检查循环依赖（简单 DFS）
    graph = {a.id: [] for a in self._agents}
    for link in self._links:
        if link.agent_id in graph:
            graph[link.agent_id].append(link.next_agent_id)

    visited = set()
    def dfs(node, path):
        if node in path:
            cycle = path[path.index(node):] + [node]
            errors.append(f"检测到循环: {' → '.join(cycle)}")
            return
        if node in visited:
            return
        path.append(node)
        for next_id in graph.get(node, []):
            dfs(next_id, path)
        path.pop()
        visited.add(node)

    for agent_id in graph:
        dfs(agent_id, [])

    return errors

def _on_validate(self) -> None:
    """验证按钮回调"""
    errors = self._validate_chain()
    if errors:
        QMessageBox.warning(self, "链验证失败", "\n".join(f"• {e}" for e in errors))
    else:
        QMessageBox.information(self, "链验证通过", "链结构有效，无错误。")
```

**验收标准**:
- [ ] 检测孤立 Agent
- [ ] 检测悬空连接
- [ ] 检测循环依赖

```python
def keyPressEvent(self, event):
    """拦截上下键浏览历史"""
    if event.key() == Qt.Key.Key_Up and self._input_history:
        self._history_index = max(0, self._history_index - 1)
        self._input_edit.setText(self._input_history[self._history_index])
    elif event.key() == Qt.Key.Key_Down and self._input_history:
        self._history_index = min(len(self._input_history), self._history_index + 1)
        if self._history_index < len(self._input_history):
            self._input_edit.setText(self._input_history[self._history_index])
        else:
            self._input_edit.clear()
    else:
        super().keyPressEvent(event)
```

```python
def _export_report(self) -> None:
    """导出评估报告"""
    from PyQt6.QtWidgets import QFileDialog

                "expected": case.expected_output,
```

```python
def _package_project(self) -> None:
    """打包项目为 ZIP"""
    import zipfile
    from presentation.project.manager import project_manager

    if not project_manager.is_open:
        QMessageBox.warning(self, "提示", "请先打开一个项目")
        return

    path = project_manager.project_path
    name = project_manager.current_project.name

    save_path, _ = QFileDialog.getSaveFileName(
        self, "保存打包文件", f"{name}.zip", "ZIP (*.zip)")
    if not save_path:
        return

    try:
        with zipfile.ZipFile(save_path, 'w', zipfile.ZIP_DEFLATED) as zf:
            for file in path.rglob("*"):
                if file.is_file() and not file.name.endswith('.pyc'):
                    zf.write(file, file.relative_to(path.parent))

        QMessageBox.information(self, "打包成功", f"项目已打包到:\n{save_path}")
    except Exception as e:
        QMessageBox.critical(self, "打包失败", str(e))
```

**验收标准**:
- [ ] 能将项目打包为 ZIP
