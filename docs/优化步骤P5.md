```python
from presentation.widgets.search_bar import SearchBar

def _search_in_log(self, keyword: str) -> None:
    """在日志中搜索关键词"""
    if not keyword:
        return
    cursor = self._log_browser.textCursor()
    document = self._log_browser.document()

    # 使用 QTextDocument 的查找功能
    found = document.find(keyword, cursor)
    if not found.isNull():
        self._log_browser.setTextCursor(found)
        self._log_browser.ensureCursorVisible()
    else:
        # 从头开始搜索
        cursor.movePosition(cursor.MoveOperation.Start)
        found = document.find(keyword, cursor)
        if not found.isNull():
            self._log_browser.setTextCursor(found)
```

**文件**: `2workbench/presentation/ops/logger_panel/log_viewer.py`

**修改方案**:

```python
from PyQt6.QtCore import QFileSystemWatcher

def _setup_file_watcher(self) -> None:
    """设置文件变化监控"""
    self._watcher = QFileSystemWatcher()
    self._watcher.fileChanged.connect(self._on_file_changed)

def _on_file_changed(self, path: str) -> None:
    """文件变化时重新加载"""
    if self._auto_scroll:
        # 追加新内容而非重新加载
        try:
            with open(path, 'r', encoding='utf-8') as f:
                content = f.read()
            self._log_browser.setPlainText(content)
            if self._auto_scroll:
                scrollbar = self._log_browser.verticalScrollBar()
                scrollbar.setValue(scrollbar.maximum())
        except Exception:
            pass
```

```python
def _package_project(self) -> None:
    """打包项目（基础实现）"""
    from presentation.project.manager import project_manager
    import zipfile
    import tempfile

    if not project_manager.is_open:
        QMessageBox.warning(self, "提示", "请先打开一个项目")
        return

    path = project_manager.project_path
    name = project_manager.current_project.name

    # 选择保存位置
    save_path, _ = QFileDialog.getSaveFileName(
        self, "保存打包文件", f"{name}.zip", "ZIP 文件 (*.zip)"
    )
    if not save_path:
        return

    try:
        with zipfile.ZipFile(save_path, 'w', zipfile.ZIP_DEFLATED) as zf:
            for file in path.rglob("*"):
                if file.is_file() and not file.name.endswith('.pyc'):
                    arcname = file.relative_to(path.parent)
                    zf.write(file, arcname)

        QMessageBox.information(self, "打包成功", f"项目已打包到:\n{save_path}")
    except Exception as e:
        QMessageBox.critical(self, "打包失败", str(e))
```
