## Step 4.1: 实现文件编辑和保存

**问题**: F-002

**文件**: `main_window.py`

**修改方案**:

```python
def _open_text_editor(self, file_path: str) -> None:
    """打开文件编辑器（可编辑模式）"""
    from PyQt6.QtWidgets import QFileDialog

    # 检查文件是否存在
    path = Path(file_path)
    if not path.exists():
        self._show_message(f"文件不存在: {file_path}")
        return

    try:
        content = path.read_text(encoding="utf-8")
    except Exception as e:
        self._show_message(f"读取文件失败: {e}")
        return

    editor = QTextEdit()
    editor.setPlainText(content)
    editor.setFont(QFont("Consolas", 11))
    editor.setReadOnly(False)  # 可编辑

    # 保存快捷键
    def save_file():
        try:
            path.write_text(editor.toPlainText(), encoding="utf-8")
            self._show_message(f"已保存: {path.name}", 2000)
        except Exception as e:
            QMessageBox.critical(self, "保存失败", str(e))

    # Ctrl+S 保存当前文件
    editor._save_callback = save_file
    QShortcut(QKeySequence("Ctrl+S"), editor).activated.connect(save_file)

    # 添加标签页
    tab_name = path.name
    index = self.center_panel.tab_widget.addTab(editor, tab_name)
    self.center_panel.tab_widget.setCurrentIndex(index)
    self.center_panel.remove_welcome_tab()
```

**验收标准**:
- [ ] 双击文件可编辑
- [ ] Ctrl+S 保存文件内容
- [ ] 保存成功有提示

## Step 4.2: 实现撤销/重做

**问题**: F-003

**文件**: `main_window.py`

**修改方案**:

QTextEdit 自带撤销/重做功能，只需确保快捷键不被拦截：

```python
# 在 _setup_menu 中，移除或禁用空的撤销/重做 action
# 让 QTextEdit 使用默认的 Ctrl+Z / Ctrl+Y 行为

# 如果需要跨编辑器的撤销栈，使用 QUndoStack：
from PyQt6.QtGui import QUndoStack, QUndoCommand

class TextEditCommand(QUndoCommand):
    def __init__(self, editor, old_text, new_text, description="编辑"):
        super().__init__(description)
        self._editor = editor
        self._old_text = old_text
        self._new_text = new_text

    def undo(self):
        self._editor.setPlainText(self._old_text)

    def redo(self):
        self._editor.setPlainText(self._new_text)
```

**验收标准**:

import hashlib
import secrets

class GUIControlHandler(http.server.BaseHTTPRequestHandler):
    # 生成随机 token
    _auth_token = secrets.token_hex(32)

    def _check_auth(self) -> bool:
        """检查请求中的认证 token"""
        auth = self.headers.get("X-Auth-Token", "")
        return auth == self._auth_token

    def do_GET(self):
        if not self._check_auth():
            self._send_json({"error": "Unauthorized"}, 401)
            return
        # ... 原有逻辑

    def do_POST(self):
        if not self._check_auth():
            self._send_json({"error": "Unauthorized"}, 401)
            return
        # ... 原有逻辑

    @classmethod
    def get_auth_token(cls) -> str:
        return cls._auth_token

## Step 4.4: 跨平台截图支持

**问题**: F-019

**文件**: `server.py`

**修改方案**:

```python
def _capture_window(self) -> bytes | None:
    """跨平台窗口截图"""
    import sys

    if sys.platform == "win32":
        return self._capture_window_winapi()
    else:
        return self._capture_with_qt()

def _capture_with_qt(self) -> bytes | None:
    """Qt 方式截图（跨平台）"""
    from PyQt6.QtWidgets import QApplication
    from PyQt6.QtGui import QPixmap

    widget = self.gui_instance
    pixmap = widget.grab()
    buffer = QByteArray()
    buffer.open(QIODevice.OpenModeFlag.ReadWrite)
    pixmap.save(buffer, "PNG")
    return bytes(buffer)
```

**验收标准**:
